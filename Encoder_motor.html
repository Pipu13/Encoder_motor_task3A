  <div class="container">

    <h2>Motor Speed Control</h2>

    <button onclick="connect()">Connect Bluetooth</button>

    <label for="speed">Set Speed (-255 to 255):</label>

    <input type="range" id="speed" min="-255" max="255" value="0" oninput="updateSpeedLabel()" />

    <div class="value" id="speedLabel">0</div>

    <button onclick="sendSpeed()">Send Speed</button>

    <h3>Current Speed: <span id="currentSpeed">--</span></h3>

  </div>


  <script>

    let device, characteristic;


    async function connect() {

      try {

        device = await navigator.bluetooth.requestDevice({

          filters: [{ namePrefix: 'ESP32' }],

          optionalServices: ['0000ffe0-0000-1000-8000-00805f9b34fb']

        });

        const server = await device.gatt.connect();

        const service = await server.getPrimaryService('0000ffe0-0000-1000-8000-00805f9b34fb');

        characteristic = await service.getCharacteristic('0000ffe1-0000-1000-8000-00805f9b34fb');

        characteristic.startNotifications();

        characteristic.addEventListener("characteristicvaluechanged", handleIncoming);

        alert("Connected to ESP32!");

      } catch (error) {

        alert("Failed to connect: " + error);

      }

    }


    function updateSpeedLabel() {

      const val = document.getElementById("speed").value;

      document.getElementById("speedLabel").innerText = val;

    }


    function sendSpeed() {

      if (!characteristic) {

        alert("Connect first!");

        return;

      }

      const speed = document.getElementById("speed").value;

      const encoder = new TextEncoder();

      characteristic.writeValue(encoder.encode(`SPD:${speed}\n`));

    }


    function handleIncoming(event) {

      const decoder = new TextDecoder();

      const value = decoder.decode(event.target.value);

      if (value.startsWith("CUR:")) {

        document.getElementById("currentSpeed").innerText = value.substring(4).trim();

      }

    }

  </script>
